#!/usr/bin/env python3 """ patch_cleanup.py — apply exactly two dead-code cleanups to index.html. 1. Remove the unused CSS rule: .collapsed .case-collapsible{display:none} 2. Remove the inert `feed:true,` / `feed:false,` properties from the 12 SOURCES entries (renderSources() ignores them; live-vs-link is decided by whether feeds.json returns items). Nothing else is touched. The brand line and its suffixes are left as-is. Usage: python3 patch_cleanup.py index.html # writes index.html in place (after backup) python3 patch_cleanup.py index.html out.html # writes a separate output file The script REFUSES to write if the expected original text isn't found, so it can never silently corrupt the file. """ import sys import pathlib import shutil CSS_TARGET = ".collapsed .case-collapsible{display:none}\n" FEED_TOKENS = ("feed:true, ", "feed:false, ") def main(): if len(sys.argv) < 2: print("usage: python3 patch_cleanup.py index.html [out.html]", file=sys.stderr) sys.exit(2) src = pathlib.Path(sys.argv[1]) dst = pathlib.Path(sys.argv[2]) if len(sys.argv) > 2 else src if not src.exists(): print(f"ERROR: {src} not found", file=sys.stderr) sys.exit(1) text = src.read_text(encoding="utf-8") original_len = len(text) # --- Edit 1: the dead CSS rule --- css_hits = text.count(".collapsed .case-collapsible") if css_hits == 0: print("ERROR: could not find '.collapsed .case-collapsible' rule. " "File may already be patched or is not the expected index.html.", file=sys.stderr) sys.exit(1) if css_hits > 1: print(f"ERROR: found {css_hits} occurrences of the .collapsed rule; expected 1. " "Aborting to avoid an ambiguous edit.", file=sys.stderr) sys.exit(1) # Remove the rule whether or not it ends with a newline. if CSS_TARGET in text: text = text.replace(CSS_TARGET, "", 1) else: # Fallback: remove just the rule text, leaving surrounding whitespace intact. text = text.replace(".collapsed .case-collapsible{display:none}", "", 1) # --- Edit 2: the inert feed:* properties in SOURCES --- feed_true = text.count(FEED_TOKENS[0]) feed_false = text.count(FEED_TOKENS[1]) total_feed = feed_true + feed_false if total_feed == 0: print("WARNING: no 'feed:true,'/'feed:false,' tokens found. " "SOURCES may already be cleaned — leaving as-is.", file=sys.stderr) else: text = text.replace(FEED_TOKENS[0], "").replace(FEED_TOKENS[1], "") # --- Post-edit assertions --- problems = [] if ".collapsed .case-collapsible" in text: problems.append("CSS rule still present after edit") if "feed:true" in text or "feed:false" in text: problems.append("feed:* property still present after edit") if problems: print("ERROR: post-edit verification failed: " + "; ".join(problems), file=sys.stderr) sys.exit(1) # --- Backup + write --- if dst == src: backup = src.with_suffix(src.suffix + ".bak") shutil.copy2(src, backup) print(f"Backup written: {backup}") dst.write_text(text, encoding="utf-8") print("Patched successfully.") print(f" CSS rule removed: 1") print(f" feed:true properties removed: {feed_true}") print(f" feed:false properties removed: {feed_false}") print(f" bytes: {original_len:,} -> {len(text):,} ({original_len - len(text)} removed)") print(f" output: {dst}") if __name__ == "__main__": main()
